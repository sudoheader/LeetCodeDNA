$O(n)$ Time Complexity
```java
class Solution {
	public int calPoints(String[] ops) {
        Deque<Integer> stack = new ArrayDeque<>();
        int result = 0;
        
        for(String op : ops) {
            if(op.equals("C")) {
                result -= stack.pop();
            } else if(op.equals("D")) {
                result += stack.peek() * 2;
                stack.push(stack.peek() * 2);
            } else if(op.equals("+")) {
                int top1 = stack.pop();
                int top2 = stack.peek();
                result += top1 + top2;
                stack.push(top1);
                stack.push(top1 + top2);
            } else {
                stack.push(Integer.parseInt(op));
                result += Integer.parseInt(op);
            }
        }
        return result;
	}
}
```

```java
class Solution {
	public int calPoints(String[] ops) {
		int sum = 0;
        LinkedList<Integer> list = new LinkedList<>();
        for(String op : ops) {
            if(op.equals("C")) {
                sum -= list.removeLast();
            } else if(op.equals("D")) {
                list.add(list.peekLast() * 2);
                sum += list.peekLast();
            } else if(op.equals("+")) {
                list.add(list.peekLast() + list.get(list.size() - 2));
                sum += list.peekLast();
            } else {
                list.add(Integer.parseInt(op));
                sum += list.peekLast();
            }
        }
        return sum;
	}
}
```
